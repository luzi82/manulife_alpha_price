function closeBrowserAlert() {
    var expiredays = 1000;
    var exdate = new Date();
    exdate.setDate(exdate.getDate()+expiredays);
    document.cookie = "closeBrowserAlert=" +escape(true)+ ';path=/'+ ";expires="+exdate.toUTCString();
    $("#browser_alert").hide();
    
}
function isClosedBrowserAlert() {
    var cookieName = "closeBrowserAlert";
    if (document.cookie.length>0) {
        start = document.cookie.indexOf(cookieName + "=");
        if (start!=-1) {
            start = start + cookieName.length+1;
            end = document.cookie.indexOf(";", start);
            if (end == -1) end=document.cookie.length;
            
            return unescape(document.cookie.substring(start, end)) == "true";
        }
    }
    return false;
}
